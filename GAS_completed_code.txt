const SHEET_ID = '12yZfMnBIqW5Q7WOtHnnZLIDz9NAgkTem8RFoTMINUfU';

// JST日時文字列を返すヘルパー（スプレッドシートへの保存時に統一使用）
function nowJST() {
  return Utilities.formatDate(new Date(), "JST", "yyyy/MM/dd HH:mm:ss");
}

// JST日付のみ（時刻なし）を返すヘルパー
function todayJST() {
  return Utilities.formatDate(new Date(), "JST", "yyyy/MM/dd");
}

// 操作ログシートを取得（なければ自動生成）するヘルパー
function getOrCreateAuditLogSheet(ss) {
  let logSheet = ss.getSheetByName('操作ログ');
  if (!logSheet) {
    logSheet = ss.insertSheet('操作ログ');
    logSheet.appendRow(['日時', '操作種別', '対象顧客', '元の担当スタッフ', '操作内容詳細']);
  }
  return logSheet;
}

// 電話番号を正規化するヘルパー（ハイフン・スペース・先頭のアポストロフィを除去）
function normalizePhoneGAS(phone) {
  return String(phone).replace(/[-\s']/g, '').trim();
}

// 電話番号で顧客の前払い残高を取得するヘルパー
function getCustomerBalanceByPhone(ss, phone) {
  const sheet = ss.getSheetByName("前払い管理");
  if (!sheet) return 0;
  const normalizedPhone = normalizePhoneGAS(phone);
  if (!normalizedPhone) return 0;
  const values = sheet.getDataRange().getValues();
  for (let i = 1; i < values.length; i++) {
    if (normalizePhoneGAS(values[i][1]) === normalizedPhone) {
      return Number(values[i][2]) || 0;
    }
  }
  return 0;
}

// 後方互換のため旧名でも呼べるようにする（内部では電話番号を使う）
function getCustomerBalance(ss, customerName) {
  // 名前で検索（後方互換）
  const sheet = ss.getSheetByName("前払い管理");
  if (!sheet) return 0;
  const trimmedName = String(customerName).trim();
  const values = sheet.getDataRange().getValues();
  for (let i = 1; i < values.length; i++) {
    if (String(values[i][0]).trim() === trimmedName) {
      return Number(values[i][2]) || 0;
    }
  }
  return 0;
}

function doPost(e) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  
  try {
    const data = JSON.parse(e.postData.contents);
    const action = data.action;
    const output = ContentService.createTextOutput().setMimeType(ContentService.MimeType.JSON);

    // デポジット関連の排他制御が必要なアクション一覧
    const depositLockActions = ['addReport', 'editReport', 'deleteReport', 'updateDeposit', 'fixDepositBalance', 'deleteDepositHistory', 'editDepositHistory'];
    let lock = null;
    if (depositLockActions.includes(action)) {
      lock = LockService.getDocumentLock();
      try {
        lock.waitLock(15000); // 最大15秒待機
      } catch (lockError) {
        return output.setContent(JSON.stringify({ success: false, message: 'サーバーが混み合っています。少し時間を置いて再度お試しください。' }));
      }
    }
    
    // ① 業務報告の追加（重複チェック対応）
    if (action === 'addReport') {
      const reportSheet = ss.getSheetByName('業務報告');
      const depositSheet = ss.getSheetByName('前払い管理');
      
      // 重複チェック要求の場合
      if (data.checkDuplicate && reportSheet) {
        const values = reportSheet.getDataRange().getValues();
        let isDuplicate = false;
        for (let i = 1; i < values.length; i++) {
          // 日付・スタッフ・電話番号が一致したら重複とみなす
          if (values[i][1] === data.date && values[i][2] === data.staff && values[i][3] === data.customerPhone) {
            isDuplicate = true;
            break;
          }
        }
        if (isDuplicate) {
          return output.setContent(JSON.stringify({ success: false, duplicate: true }));
        }
      }

      // 新規顧客の自動登録（業務報告の前に顧客リスト・前払い管理へ追加）
      if (data.isNewCustomer) {
        const custSheet = ss.getSheetByName('顧客リスト') || (function() {
          const s = ss.insertSheet('顧客リスト');
          s.appendRow(['お客様名', '電話番号', '登録日']);
          return s;
        })();
        // 電話番号で既存チェック（二重登録防止）
        const custValues = custSheet.getDataRange().getValues();
        let custExists = false;
        for (let i = 1; i < custValues.length; i++) {
          if (String(custValues[i][1]).replace(/[-']/g, '') === String(data.customerPhone).replace(/-/g, '')) {
            custExists = true;
            break;
          }
        }
        if (!custExists) {
          custSheet.appendRow([data.customerName, "'" + data.customerPhone, nowJST()]);
          // 前払い管理にも0円で枠を作成（★ 電話番号重複チェック付き）
          const depSheet = ss.getSheetByName('前払い管理');
          if (depSheet) {
            const depVals = depSheet.getDataRange().getValues();
            const normPhone = normalizePhoneGAS(data.customerPhone);
            let depExists = false;
            for (let di = 1; di < depVals.length; di++) {
              if (normalizePhoneGAS(depVals[di][1]) === normPhone) {
                depExists = true;
                break;
              }
            }
            if (!depExists) {
              depSheet.appendRow([data.customerName, "'" + data.customerPhone, 0]);
            }
          }
        }
      }

      const newId = Utilities.getUuid();
      let paidStatus = '未入金';
      let autoDeducted = false;
      let insufficientBalance = false;
      let currentDeposit = 0;
      let depositUsed = 0; // 実際に使用したデポジット金額
      let billingAmount = 0; // 請求金額
      const sales = Number(data.totalSales) || 0;
      const customerName = data.customerName;
      const customerPhone = normalizePhoneGAS(data.customerPhone); // ★ 電話番号で照合

      // 前払い残高から自動引き落とし（★ 電話番号で照合に変更）
      if (depositSheet && customerPhone && sales > 0) {
        const dValues = depositSheet.getDataRange().getValues();
        let customerFound = false;
        for (let i = 1; i < dValues.length; i++) {
          if (normalizePhoneGAS(dValues[i][1]) === customerPhone) {
            customerFound = true;
            currentDeposit = Number(dValues[i][2]) || 0; // 残高は3列目(C列)
            if (currentDeposit >= sales) {
              // デポジットで全額カバーできる場合
              const newBal = currentDeposit - sales;
              depositSheet.getRange(i + 1, 3).setValue(newBal);
              paidStatus = '入金済';
              autoDeducted = true;
              depositUsed = sales;
              billingAmount = 0;
              // デポジット履歴に自動引き落としを記録
              let histSheet = ss.getSheetByName('デポジット履歴');
              if (!histSheet) {
                histSheet = ss.insertSheet('デポジット履歴');
                histSheet.appendRow(['日時', 'お客様名', '電話番号', '増減額', '種別', '残高']);
              }
              histSheet.appendRow([nowJST(), customerName, normalizePhoneGAS(customerPhone), -sales, '利用(自動引落)', newBal]);
            } else if (currentDeposit > 0) {
              // デポジットで一部カバーできる場合
              depositSheet.getRange(i + 1, 3).setValue(0);
              insufficientBalance = true;
              depositUsed = currentDeposit;
              billingAmount = sales - currentDeposit;
              let histSheet = ss.getSheetByName('デポジット履歴');
              if (!histSheet) {
                histSheet = ss.insertSheet('デポジット履歴');
                histSheet.appendRow(['日時', 'お客様名', '電話番号', '増減額', '種別', '残高']);
              }
              histSheet.appendRow([nowJST(), customerName, normalizePhoneGAS(customerPhone), -currentDeposit, '利用(一部引落)', 0]);
            } else {
              depositUsed = 0;
              billingAmount = sales;
            }
            break;
          }
        }
        if (!customerFound) {
          depositUsed = 0;
          billingAmount = sales;
        }
      } else {
        depositUsed = 0;
        billingAmount = sales;
      }

      reportSheet.appendRow([
        newId,
        data.date,
        data.staff,
        `'${data.customerPhone}`,
        data.customerName,
        data.services,
        sales,
        data.staffShare,
        paidStatus,  // デポジット全額カバーの場合「入金済」、それ以外「未入金」
        '', // 入金日は空（オーナーが手動でチェックした際に記録）
        depositUsed, // デポジット使用額
        billingAmount, // 請求額
        data.memo || '' // メモ・備考（M列）
      ]);
      
      // スタッフおよびオーナーへの業務報告確認メール自動送信
      let emailSent = false;
      const ownerEmail = 'iyashi.hanashitara@gmail.com';
      const staffSheet = ss.getSheetByName('スタッフマスタ');
      let staffEmail = '';
      if (staffSheet) {
        const staffData = staffSheet.getDataRange().getValues();
        for (let j = 1; j < staffData.length; j++) {
          if (staffData[j][0] === data.staff) {
            staffEmail = String(staffData[j][2]).trim();
            break;
          }
        }
      }
      
      try {
        let formattedDate = data.date;
        if (Object.prototype.toString.call(formattedDate) === '[object Date]') {
          formattedDate = Utilities.formatDate(formattedDate, "JST", "yyyy年MM月dd日");
        }
        const emailBody = `${data.staff} 様\n\nお疲れ様です。ハナシタラ.comです。\n業務報告を受け付けました。以下が報告内容です。\n\n【対応日】${formattedDate}\n【お客様名】${data.customerName}\n【サービス内容】${data.services}\n【総売上】¥${sales.toLocaleString()}\n【スタッフ報酬】¥${Number(data.staffShare).toLocaleString()}${data.memo ? '\n【メモ】' + data.memo : ''}\n\nこの内容は控えとしてお手元に保管ください。\n引き続きよろしくお願いいたします。`;
        
        // スタッフメールが正しく登録されていればスタッフ宛＋BCCでオーナーへ
        if (staffEmail && staffEmail !== 'demo@example.com' && staffEmail.includes('@')) {
          MailApp.sendEmail(staffEmail, '【ハナシタラ.com】業務報告の受付確認', emailBody, {
            name: 'ハナシタラ.com 業務システム',
            bcc: ownerEmail
          });
          emailSent = true;
        } else {
          // スタッフメール未登録の場合はオーナー宛のみ通知
          MailApp.sendEmail(ownerEmail, `【ハナシタラ.com】業務報告の受付（${data.staff}様のメールアドレス未登録）`, emailBody, {
            name: 'ハナシタラ.com 業務システム'
          });
        }
      } catch(e) { /* メール失敗でも報告自体は成功扱い */ }

      return output.setContent(JSON.stringify({
        success: true,
        id: newId,
        autoDeducted: autoDeducted,
        insufficientBalance: insufficientBalance,
        currentDeposit: currentDeposit,
        emailSent: emailSent
      }));
    }
    
    // ② 業務報告の修正（管理者用・デポジット差分再計算付き）
    if (action === 'editReport') {
      const sheet = ss.getSheetByName('業務報告');
      const values = sheet.getDataRange().getValues();
      for(let i = 1; i < values.length; i++) {
        if(values[i][0] === data.id) {
          const customerName = data.customerName;
          const oldSales = Number(values[i][6]) || 0; // G列: 元の売上額
          const newSales = Number(data.totalSales) || 0;
          const oldDepositUsed = Number(values[i][10]) || 0; // K列: 元のデポジット使用額
          const paidStatus = values[i][8]; // I列: 入金状況

          // 操作ログ：編集前のスナップショットを記録（フェイルセーフ）
          try {
            const auditSheet = getOrCreateAuditLogSheet(ss);
            const oldCustomerName = String(values[i][4]).trim();
            const oldPhone = String(values[i][3]).trim();
            const oldStaff = String(values[i][2]).trim();
            const detailParts = [];
            if (oldSales !== newSales) {
              detailParts.push('売上金額が変更されました（変更前: ¥' + oldSales.toLocaleString() + ' → 変更後: ¥' + newSales.toLocaleString() + '）');
            }
            if (oldCustomerName !== customerName) {
              detailParts.push('顧客名が変更されました（変更前: ' + oldCustomerName + ' → 変更後: ' + customerName + '）');
            }
            if (normalizePhoneGAS(oldPhone) !== normalizePhoneGAS(data.customerPhone)) {
              detailParts.push('電話番号が変更されました（変更前: ' + oldPhone + ' → 変更後: ' + data.customerPhone + '）');
            }
            if (detailParts.length === 0) {
              detailParts.push('データが上書き保存されました（売上: ¥' + newSales.toLocaleString() + '）');
            }
            auditSheet.appendRow([nowJST(), '業務報告の編集', oldCustomerName, oldStaff, detailParts.join(' / ')]);
          } catch (logErr) { /* ログ書き込み失敗でもメイン処理を続行 */ }

          // D列(電話), E列(名前), G列(売上額), H列(スタッフ報酬) を更新
          sheet.getRange(i + 1, 4).setValue(`'${data.customerPhone}`);
          sheet.getRange(i + 1, 5).setValue(customerName);
          sheet.getRange(i + 1, 7).setValue(newSales);
          
          // 売上変更に伴うスタッフ報酬（H列）の自動比率計算
          const oldStaffShare = Number(values[i][7]) || 0;
          let newStaffShare = oldStaffShare;
          if (oldSales > 0) {
            const ratio = oldStaffShare / oldSales;
            newStaffShare = Math.round(newSales * ratio);
          } else {
            const oldStaff = String(values[i][2]).trim();
            newStaffShare = (oldStaff === '吉川（オーナー）') ? newSales : Math.round(newSales * 0.6);
          }
          sheet.getRange(i + 1, 8).setValue(newStaffShare);

          // デポジット引き落とし済みの場合、差額を再計算（★ 電話番号で照合）
          const editPhone = normalizePhoneGAS(data.customerPhone || values[i][3]);
          if (oldDepositUsed > 0 && oldSales !== newSales) {
            const depSheet = ss.getSheetByName("前払い管理");
            if (depSheet && editPhone) {
              const depValues = depSheet.getDataRange().getValues();
              for (let j = 1; j < depValues.length; j++) {
                if (normalizePhoneGAS(depValues[j][1]) === editPhone) {
                  const currentBal = Number(depValues[j][2]) || 0;

                  let histSheet = ss.getSheetByName("デポジット履歴");
                  if (newSales > oldSales) {
                    const diff = newSales - oldSales;
                    const additionalDeduct = Math.min(diff, currentBal);
                    const newBal = currentBal - additionalDeduct;
                    depSheet.getRange(j + 1, 3).setValue(newBal);
                    const newDepositUsed = oldDepositUsed + additionalDeduct;
                    sheet.getRange(i + 1, 11).setValue(newDepositUsed);
                    sheet.getRange(i + 1, 12).setValue(newSales - newDepositUsed);
                    if (histSheet && additionalDeduct > 0) {
                      histSheet.appendRow([nowJST(), customerName, editPhone, -additionalDeduct, '報告編集(追加引落)', newBal]);
                    }
                  } else {
                    const diff = oldSales - newSales;
                    const refund = Math.min(diff, oldDepositUsed);
                    const newBal = currentBal + refund;
                    depSheet.getRange(j + 1, 3).setValue(newBal);
                    const newDepositUsed = oldDepositUsed - refund;
                    sheet.getRange(i + 1, 11).setValue(newDepositUsed);
                    sheet.getRange(i + 1, 12).setValue(newSales - newDepositUsed);
                    if (histSheet && refund > 0) {
                      histSheet.appendRow([nowJST(), customerName, editPhone, refund, '報告編集(返還)', newBal]);
                    }
                  }
                  break;
                }
              }
            }
          } else if (oldDepositUsed === 0) {
            sheet.getRange(i + 1, 12).setValue(newSales);
          }

          // デポジット差額処理が走らなかった場合でも、利用履歴に編集の痕跡を残す（0円通知行）
          if (!(oldDepositUsed > 0 && oldSales !== newSales)) {
            let editHistSheet = ss.getSheetByName("デポジット履歴");
            if (editHistSheet) {
              const currentBal = getCustomerBalanceByPhone(ss, editPhone);
              const editLabel = oldSales !== newSales
                ? '報告編集(¥' + oldSales.toLocaleString() + '→¥' + newSales.toLocaleString() + ')'
                : '報告編集(顧客情報の変更)';
              editHistSheet.appendRow([nowJST(), customerName, editPhone, 0, editLabel, currentBal]);
            }
          }

          const newBalance = getCustomerBalanceByPhone(ss, editPhone);
          if (lock) lock.releaseLock();
          return output.setContent(JSON.stringify({ success: true, customerName: customerName, newBalance: newBalance }));
        }
      }
      if (lock) lock.releaseLock();
      return output.setContent(JSON.stringify({ success: false, message: 'IDが見つかりません' }));
    }

    // ③ 入金状況の更新
    if (action === 'updatePaidStatus') {
      const sheet = ss.getSheetByName('業務報告');
      const targetId = data.id;
      const newStatus = data.isPaid ? '入金済' : '未入金';
      const paymentDate = data.paymentDate || ''; // 入金日（入金チェックを押した日付）

      const values = sheet.getDataRange().getValues();
      for (let i = 1; i < values.length; i++) {
        if (values[i][0] === targetId) {
          sheet.getRange(i + 1, 9).setValue(newStatus);
          // J列（10列目）に入金日を保存
          if (data.isPaid && paymentDate) {
            sheet.getRange(i + 1, 10).setValue(paymentDate);
          } else {
            sheet.getRange(i + 1, 10).setValue(''); // 未入金の場合は空にする
          }

          // 入金済に変更した場合、スタッフにメール通知
          let emailSent = false;
          if (data.isPaid) {
            const staffName = values[i][2]; // C列: スタッフ名
            const customerName = values[i][4]; // E列: 顧客名
            const services = values[i][5]; // F列: サービス内容
            const totalSales = values[i][6]; // G列: 総売上
            const reportDate = values[i][1]; // B列: 日付
            const memo = values[i][12]; // M列: 備考

            const staffSheet = ss.getSheetByName('スタッフマスタ');
            if (staffSheet) {
              const staffData = staffSheet.getDataRange().getValues();
              let staffEmail = '';
              for (let j = 1; j < staffData.length; j++) {
                if (staffData[j][0] === staffName) {
                  staffEmail = String(staffData[j][2]).trim();
                  break;
                }
              }
              if (staffEmail && staffEmail !== 'demo@example.com') {
                try {
                  // 日付が英語表記になるのを防ぐため、Date型の場合は日本時間にフォーマットする
                  let formattedDate = reportDate;
                  if (Object.prototype.toString.call(reportDate) === '[object Date]') {
                    formattedDate = Utilities.formatDate(reportDate, "JST", "yyyy年MM月dd日");
                  }

                  const emailBody = `${staffName} 様\n\nお疲れ様です。ハナシタラ.comです。\n入金が確認されました。\n\n対応日: ${formattedDate}\nお客様名: ${customerName}\nサービス内容: ${services}\n売上金額: ¥${Number(totalSales).toLocaleString()}\n備考: ${memo || 'なし'}\n\nお客様へ入金確認の旨をお伝えください。\nよろしくお願いいたします。`;
                  MailApp.sendEmail(staffEmail, '【ハナシタラ.com】入金確認のお知らせ', emailBody, {
                    name: 'ハナシタラ.com 業務システム'
                  });
                  emailSent = true;
                } catch(e) { /* メール失敗でも入金更新は成功扱い */ }
              }
            }
          }

          return output.setContent(JSON.stringify({ success: true, emailSent: emailSent }));
        }
      }
      return output.setContent(JSON.stringify({ success: false, message: 'IDが見つかりません' }));
    }

    // ④ ブラックリストの追加
    if (action === 'addBlacklist') {
      const sheet = ss.getSheetByName('ブラックリスト');
      sheet.appendRow([
        `'${data.phone}`,
        data.name,
        data.reason,
        nowJST()
      ]);

      // 全スタッフへブラックリスト登録の一斉通知メール
      let emailsSent = 0;
      const staffSheet = ss.getSheetByName('スタッフマスタ');
      if (staffSheet) {
        const staffData = staffSheet.getDataRange().getValues();
        const emailBody = `スタッフ各位\n\nお疲れ様です。ハナシタラ.comです。\n以下のお客様がブラックリストに登録されました。\n\n【電話番号】${data.phone}\n【お客様名】${data.name}\n【理由】${data.reason}\n\nこの番号からのお電話はお断りしてください。\nよろしくお願いいたします。`;
        for (let j = 1; j < staffData.length; j++) {
          const email = String(staffData[j][2]).trim();
          if (email && email !== 'demo@example.com') {
            try {
              MailApp.sendEmail(email, '【ハナシタラ.com】ブラックリスト登録のお知らせ', emailBody, {
                name: 'ハナシタラ.com 業務システム'
              });
              emailsSent++;
            } catch(e) { /* 個別のメール失敗はスキップ */ }
          }
        }
      }

      return output.setContent(JSON.stringify({ success: true, emailsSent: emailsSent }));
    }

    // ⑤ お客様の新規追加（電話番号対応・顧客リストへ）
    if (action === 'addCustomer') {
      let sheet = ss.getSheetByName("顧客リスト");
      if (!sheet) {
        sheet = ss.insertSheet("顧客リスト");
        sheet.appendRow(["お客様名", "電話番号", "登録日", "パスワード"]);
      }
      
      const customerName = data.customerName;
      const phone = data.customerPhone || '';
      const values = sheet.getDataRange().getValues();
      
      let exists = false;
      const normNewPhone = normalizePhoneGAS(phone);
      for (let i = 1; i < values.length; i++) {
        // 名前 OR 電話番号 で重複チェック
        if (values[i][0] === customerName || (normNewPhone && normalizePhoneGAS(values[i][1]) === normNewPhone)) {
          exists = true;
          break;
        }
      }
      
      if (!exists) {
        sheet.appendRow([customerName, `'${phone}`, nowJST(), data.password || '']);
      }
      
      // デポジット用にも0円で枠を作っておく（★ 電話番号重複チェック付き）
      let depSheet = ss.getSheetByName("前払い管理");
      if (depSheet) {
        const depValues = depSheet.getDataRange().getValues();
        const normPhone = normalizePhoneGAS(phone);
        let depExists = false;
        for (let i = 1; i < depValues.length; i++) {
          // 電話番号 OR 名前 で重複チェック
          if ((normPhone && normalizePhoneGAS(depValues[i][1]) === normPhone) || String(depValues[i][0]).trim() === customerName) {
            depExists = true;
            break;
          }
        }
        if (!depExists) depSheet.appendRow([customerName, `'${phone}`, 0]);
      }

      return output.setContent(JSON.stringify({ success: true }));
    }

    // ⑤-1.5 お客様の名前・電話情報の修正
    if (action === 'editCustomer') {
      const oldName = data.oldName;
      const newName = data.newName;
      const phone = data.phone;
      
      // 1. 顧客リストの更新
      const custSheet = ss.getSheetByName("顧客リスト");
      if (custSheet) {
        const values = custSheet.getDataRange().getValues();
        for (let i = 1; i < values.length; i++) {
          if (values[i][0] === oldName) {
             custSheet.getRange(i + 1, 1).setValue(newName);
             custSheet.getRange(i + 1, 2).setValue(`'${phone}`);
             if (data.password !== undefined) custSheet.getRange(i + 1, 4).setValue(data.password);
          }
        }
      }

      // 2. 前払い管理の更新
      const depSheet = ss.getSheetByName("前払い管理");
      if (depSheet) {
        const values = depSheet.getDataRange().getValues();
        for (let i = 1; i < values.length; i++) {
          if (values[i][0] === oldName) {
             depSheet.getRange(i + 1, 1).setValue(newName);
             depSheet.getRange(i + 1, 2).setValue(`'${phone}`);
          }
        }
      }

      // 3. 業務報告の名前・電話番号も一貫性のために更新
      const reportSheet = ss.getSheetByName("業務報告");
      if (reportSheet) {
        const values = reportSheet.getDataRange().getValues();
        for (let i = 1; i < values.length; i++) {
          if (values[i][4] === oldName) {
             reportSheet.getRange(i + 1, 4).setValue(`'${phone}`);
             reportSheet.getRange(i + 1, 5).setValue(newName);
          }
        }
      }
      
      return output.setContent(JSON.stringify({ success: true }));
    }

    // ⑤-1.6 お客様の削除（顧客リスト・前払い管理・デポジット履歴から削除）
    if (action === 'deleteCustomer') {
      const customerName = data.customerName;
      let deletedFrom = [];

      // 1. 顧客リストから削除
      const custSheet = ss.getSheetByName("顧客リスト");
      if (custSheet) {
        const values = custSheet.getDataRange().getValues();
        for (let i = values.length - 1; i >= 1; i--) {
          if (values[i][0] === customerName) {
            custSheet.deleteRow(i + 1);
            deletedFrom.push('顧客リスト');
            break;
          }
        }
      }

      // 2. 前払い管理から削除
      const depSheet = ss.getSheetByName("前払い管理");
      if (depSheet) {
        const values = depSheet.getDataRange().getValues();
        for (let i = values.length - 1; i >= 1; i--) {
          if (values[i][0] === customerName) {
            depSheet.deleteRow(i + 1);
            deletedFrom.push('前払い管理');
            break;
          }
        }
      }

      // 3. デポジット履歴から該当顧客の行をすべて削除（★ 電話番号 or 名前で照合）
      const histSheet = ss.getSheetByName("デポジット履歴");
      if (histSheet) {
        // 顧客リストから電話番号を取得（既に削除済みの場合があるので前払い管理からも探す）
        let delCustPhone = '';
        const custListForDel = ss.getSheetByName("顧客リスト");
        if (custListForDel) {
          const clv = custListForDel.getDataRange().getValues();
          for (let ci = 1; ci < clv.length; ci++) {
            if (String(clv[ci][0]).trim() === customerName) { delCustPhone = normalizePhoneGAS(clv[ci][1]); break; }
          }
        }
        if (!delCustPhone) {
          const dpForDel = ss.getSheetByName("前払い管理");
          if (dpForDel) {
            const dpv = dpForDel.getDataRange().getValues();
            for (let di = 1; di < dpv.length; di++) {
              if (String(dpv[di][0]).trim() === customerName) { delCustPhone = normalizePhoneGAS(dpv[di][1]); break; }
            }
          }
        }
        const values = histSheet.getDataRange().getValues();
        let count = 0;
        for (let i = values.length - 1; i >= 1; i--) {
          const rowPhone = normalizePhoneGAS(values[i][2]); // C列: 電話番号
          const rowName = String(values[i][1]).trim();
          if ((delCustPhone && rowPhone === delCustPhone) || rowName === customerName) {
            histSheet.deleteRow(i + 1);
            count++;
          }
        }
        if (count > 0) deletedFrom.push('デポジット履歴(' + count + '件)');
      }

      // 注意: 業務報告は削除しない（売上・スタッフ報酬の記録として残す）
      return output.setContent(JSON.stringify({ success: true, deletedFrom: deletedFrom }));
    }

    // ⑤-2 業務報告の削除（デポジット逆戻し＋残高返却付き）
    if (action === 'deleteReport') {
      const sheet = ss.getSheetByName("業務報告");
      if (!sheet) {
        if (lock) lock.releaseLock();
        return output.setContent(JSON.stringify({ success: false, message: '業務報告シートがありません' }));
      }
      const values = sheet.getDataRange().getValues();
      for (let i = 1; i < values.length; i++) {
        if (values[i][0] === data.id) {
          const customerName = String(values[i][4]).trim(); // E列: 顧客名
          const delPhone = normalizePhoneGAS(values[i][3]); // D列: 電話番号 ★
          const depositUsed = Number(values[i][10]) || 0; // K列: デポジット使用額
          const paidStatus = values[i][8]; // I列: 入金状況

          // デポジットが使われていた場合、その分を残高に戻す（★ 電話番号で照合）
          if (depositUsed > 0 && delPhone) {
            const depSheet = ss.getSheetByName("前払い管理");
            if (depSheet) {
              const depValues = depSheet.getDataRange().getValues();
              for (let j = 1; j < depValues.length; j++) {
                if (normalizePhoneGAS(depValues[j][1]) === delPhone) {
                  const currentBal = Number(depValues[j][2]) || 0;
                  const newBal = currentBal + depositUsed;
                  depSheet.getRange(j + 1, 3).setValue(newBal);
                  break;
                }
              }
            }
            let historySheet = ss.getSheetByName("デポジット履歴");
            if (historySheet) {
              const newBal = getCustomerBalanceByPhone(ss, delPhone);
              const deletedSales = Number(values[i][6]) || 0;
              historySheet.appendRow([nowJST(), customerName, delPhone, depositUsed, '報告削除による返還(¥' + deletedSales.toLocaleString() + 'の報告)', newBal]);
            }
          } else {
            // デポジット不使用の報告でも、利用履歴に削除の痕跡を残す（0円通知行）
            let historySheet = ss.getSheetByName("デポジット履歴");
            if (historySheet) {
              const currentBal = getCustomerBalanceByPhone(ss, delPhone);
              const deletedSales = Number(values[i][6]) || 0;
              const deleteLabel = '報告削除(¥' + deletedSales.toLocaleString() + 'の報告)';
              historySheet.appendRow([nowJST(), customerName, delPhone, 0, deleteLabel, currentBal]);
            }
          }

          // 操作ログ：削除前のスナップショットを記録（フェイルセーフ）
          try {
            const auditSheet = getOrCreateAuditLogSheet(ss);
            const oldSalesVal = Number(values[i][6]) || 0;
            const oldDepositVal = Number(values[i][10]) || 0;
            const oldStaff = String(values[i][2]).trim();
            const detail = '売上 ¥' + oldSalesVal.toLocaleString() + ', デポジット使用額 ¥' + oldDepositVal.toLocaleString() + ' の報告がダッシュボードから削除されました';
            auditSheet.appendRow([nowJST(), '業務報告の削除', customerName, oldStaff, detail]);
          } catch (logErr) { /* ログ書き込み失敗でもメイン処理を続行 */ }

          sheet.deleteRow(i + 1);
          const newBalance = getCustomerBalanceByPhone(ss, delPhone);
          if (lock) lock.releaseLock();
          return output.setContent(JSON.stringify({ success: true, customerName: customerName, newBalance: newBalance }));
        }
      }
      if (lock) lock.releaseLock();
      return output.setContent(JSON.stringify({ success: false, message: '対象の報告が見つかりません' }));
    }

    // ⑥ チャージ・利用（ボーナス対応 ＆ 履歴保存 ＆ 未払い自動精算）
    if (action === 'updateDeposit') {
      const sheet = ss.getSheetByName("前払い管理");
      let historySheet = ss.getSheetByName("デポジット履歴");
      if (!historySheet) {
        historySheet = ss.insertSheet("デポジット履歴");
        historySheet.appendRow(["日時", "お客様名", "電話番号", "増減額", "種別", "残高"]);
      }

      if (!sheet) {
        if (lock) lock.releaseLock();
        return output.setContent(JSON.stringify({ success: false, message: '前払い管理シートがありません' }));
      }

      const customerName = String(data.customerName).trim();
      const chargePhone = normalizePhoneGAS(data.customerPhone); // ★ 電話番号で照合
      const amount = Number(data.amount) || 0;
      const isCharge = data.type === 'charge';

      const values = sheet.getDataRange().getValues();
      let customerRowIndex = -1;
      let currentBalance = 0;

      // ★ 電話番号でデポジット行を検索
      for (let i = 1; i < values.length; i++) {
        if (normalizePhoneGAS(values[i][1]) === chargePhone) {
          customerRowIndex = i + 1;
          currentBalance = Number(values[i][2]) || 0;
          break;
        }
      }

      let remainingCharge = amount;
      let totalSettled = 0;
      const settledReports = [];

      // チャージ時のみ：未払い業務報告を古い順に自動精算（★ 電話番号で照合）
      if (isCharge && remainingCharge > 0) {
        const reportSheet = ss.getSheetByName("業務報告");
        if (reportSheet) {
          const repValues = reportSheet.getDataRange().getValues();
          const unpaidRows = [];
          for (let i = 1; i < repValues.length; i++) {
            if (normalizePhoneGAS(repValues[i][3]) === chargePhone && repValues[i][8] === '未入金') {
              const billing = Number(repValues[i][11]) || 0;
              if (billing > 0) {
                unpaidRows.push({ rowIndex: i + 1, originalIndex: i, billing: billing, date: repValues[i][1] });
              }
            }
          }
          // 古い順にソート（行番号が小さい=古い）
          unpaidRows.sort((a, b) => a.rowIndex - b.rowIndex);

          for (const unpaid of unpaidRows) {
            if (remainingCharge <= 0) break;
            if (remainingCharge >= unpaid.billing) {
              // 全額充当
              remainingCharge -= unpaid.billing;
              totalSettled += unpaid.billing;
              reportSheet.getRange(unpaid.rowIndex, 9).setValue('入金済');
              reportSheet.getRange(unpaid.rowIndex, 10).setValue(nowJST());
              // K列: 元のデポジット使用額に充当額を加算
              const oldDepUsed = Number(repValues[unpaid.originalIndex][10]) || 0;
              reportSheet.getRange(unpaid.rowIndex, 11).setValue(oldDepUsed + unpaid.billing);
              reportSheet.getRange(unpaid.rowIndex, 12).setValue(0); // 請求額を0に
              settledReports.push({ date: unpaid.date, amount: unpaid.billing });
            } else {
              // 一部充当（残りのチャージ額で部分的に支払い）
              const partialPay = remainingCharge;
              totalSettled += partialPay;
              const oldDepUsed2 = Number(repValues[unpaid.originalIndex][10]) || 0;
              reportSheet.getRange(unpaid.rowIndex, 11).setValue(oldDepUsed2 + partialPay);
              reportSheet.getRange(unpaid.rowIndex, 12).setValue(unpaid.billing - partialPay);
              remainingCharge = 0;
              settledReports.push({ date: unpaid.date, amount: partialPay, partial: true });
            }
          }
        }
      }

      // 残ったチャージ額を残高に加算
      let newBalance = currentBalance + (isCharge ? remainingCharge : amount);
      if (customerRowIndex !== -1) {
        sheet.getRange(customerRowIndex, 3).setValue(newBalance);
      } else {
        // ★ 新規追加前に電話番号で既存行を再チェック（重複防止）
        const recheckValues = sheet.getDataRange().getValues();
        let foundExisting = false;
        for (let ri = 1; ri < recheckValues.length; ri++) {
          if (normalizePhoneGAS(recheckValues[ri][1]) === chargePhone) {
            // 既存行が見つかった場合はそちらを更新
            sheet.getRange(ri + 1, 3).setValue(Number(recheckValues[ri][2] || 0) + (isCharge ? remainingCharge : amount));
            newBalance = Number(recheckValues[ri][2] || 0) + (isCharge ? remainingCharge : amount);
            foundExisting = true;
            break;
          }
        }
        if (!foundExisting) {
          sheet.appendRow([customerName, "'" + (data.customerPhone || ''), isCharge ? remainingCharge : amount]);
          newBalance = isCharge ? remainingCharge : amount;
        }
      }

      // ★ Bug#4修正: 履歴に正確な残高を記録
      const typeLabel = isCharge ? 'チャージ' : '利用';
      if (isCharge && totalSettled > 0) {
        // チャージ＋自動精算があった場合：チャージ時点の残高と精算後の残高を分けて記録
        const balBeforeSettle = currentBalance + amount; // チャージ直後（精算前）の仮残高
        historySheet.appendRow([nowJST(), customerName, chargePhone, amount, typeLabel, balBeforeSettle]);
        historySheet.appendRow([nowJST(), customerName, chargePhone, -totalSettled, '未払い充当', newBalance]);
      } else {
        // チャージのみ or 利用のみの場合
        historySheet.appendRow([nowJST(), customerName, chargePhone, isCharge ? amount : -Math.abs(amount), typeLabel, newBalance]);
      }

      if (lock) lock.releaseLock();
      return output.setContent(JSON.stringify({
        success: true,
        newBalance: newBalance,
        autoSettled: totalSettled,
        settledReports: settledReports
      }));
    }

    // ⑥-2 残高強制上書き
    if (action === 'fixDepositBalance') {
      const sheet = ss.getSheetByName("前払い管理");
      let historySheet = ss.getSheetByName("デポジット履歴");
      if (!historySheet) {
        historySheet = ss.insertSheet("デポジット履歴");
        historySheet.appendRow(["日時", "お客様名", "電話番号", "増減額", "種別", "残高"]);
      }
      if (!sheet) {
        if (lock) lock.releaseLock();
        return output.setContent(JSON.stringify({ success: false, message: '前払い管理シートがありません' }));
      }

      const customerName = String(data.customerName).trim();
      const fixPhone = normalizePhoneGAS(data.customerPhone); // ★ 電話番号で照合
      const newBalance = Number(data.newBalance) || 0;
      const values = sheet.getDataRange().getValues();
      let found = false;
      let oldBalance = 0;

      // ★ 電話番号で検索（フォールバック: 名前で検索）
      for (let i = 1; i < values.length; i++) {
        if ((fixPhone && normalizePhoneGAS(values[i][1]) === fixPhone) || (!fixPhone && String(values[i][0]).trim() === customerName)) {
          oldBalance = Number(values[i][2]) || 0;
          sheet.getRange(i + 1, 3).setValue(newBalance);
          found = true;
          break;
        }
      }

      if (!found) {
        if (lock) lock.releaseLock();
        return output.setContent(JSON.stringify({ success: false, message: '顧客が見つかりません' }));
      }

      const diff = newBalance - oldBalance;
      historySheet.appendRow([nowJST(), customerName, fixPhone, diff, '残高調整', newBalance]);

      if (lock) lock.releaseLock();
      return output.setContent(JSON.stringify({ success: true, newBalance: newBalance }));
    }

    // ⑥-3 デポジット履歴の個別削除（残高自動連動）
    if (action === 'deleteDepositHistory') {
      const historySheet = ss.getSheetByName("デポジット履歴");
      if (!historySheet) {
        if (lock) lock.releaseLock();
        return output.setContent(JSON.stringify({ success: false, message: 'デポジット履歴シートがありません' }));
      }

      const targetCustomerName = String(data.customerName).trim();
      const delHistPhone = normalizePhoneGAS(data.customerPhone);
      const values = historySheet.getDataRange().getValues();
      for (let i = 1; i < values.length; i++) {
        const rowDate = values[i][0];
        const rowName = String(values[i][1]).trim();
        const rowPhone = normalizePhoneGAS(values[i][2]); // C列: 電話番号
        const rowAmount = Number(values[i][3]); // D列: 増減額
        const rowType = String(values[i][4]); // E列: 種別

        // 日付比較: スプレッドシートがDate型で保持している場合はgetTime()で比較
        let dateMatch = false;
        if (rowDate instanceof Date && data.date) {
          const inputDate = new Date(data.date);
          dateMatch = !isNaN(inputDate.getTime()) && rowDate.getTime() === inputDate.getTime();
        } else {
          dateMatch = String(rowDate) === String(data.date);
        }

        // ★ 電話番号で照合（フォールバック: 名前照合）
        const customerMatch = (delHistPhone && rowPhone) ? (rowPhone === delHistPhone) : (rowName === targetCustomerName);
        if (dateMatch && customerMatch && rowAmount === Number(data.amount) && rowType === data.type) {
          // 残高自動連動: 削除する履歴のamount分を残高から逆算（★ 電話番号で照合）
          const depSheet = ss.getSheetByName("前払い管理");
          if (depSheet && rowAmount !== 0) {
            const depValues = depSheet.getDataRange().getValues();
            for (let j = 1; j < depValues.length; j++) {
              if ((delHistPhone && normalizePhoneGAS(depValues[j][1]) === delHistPhone) || (!delHistPhone && String(depValues[j][0]).trim() === targetCustomerName)) {
                const currentBal = Number(depValues[j][2]) || 0;
                depSheet.getRange(j + 1, 3).setValue(currentBal - rowAmount);
                break;
              }
            }
          }

          historySheet.deleteRow(i + 1);
          const newBalance = delHistPhone ? getCustomerBalanceByPhone(ss, delHistPhone) : getCustomerBalance(ss, targetCustomerName);
          if (lock) lock.releaseLock();
          return output.setContent(JSON.stringify({ success: true, newBalance: newBalance }));
        }
      }

      if (lock) lock.releaseLock();
      return output.setContent(JSON.stringify({ success: false, message: '該当する履歴が見つかりません' }));
    }

    // ⑥-4 デポジット履歴の金額編集（残高自動連動）
    if (action === 'editDepositHistory') {
      const historySheet = ss.getSheetByName("デポジット履歴");
      if (!historySheet) {
        if (lock) lock.releaseLock();
        return output.setContent(JSON.stringify({ success: false, message: 'デポジット履歴シートがありません' }));
      }

      const targetCustomerName = String(data.customerName).trim();
      const editHistPhone = normalizePhoneGAS(data.customerPhone);
      const values = historySheet.getDataRange().getValues();
      for (let i = 1; i < values.length; i++) {
        const rowDate = values[i][0];
        const rowName = String(values[i][1]).trim();
        const rowPhone = normalizePhoneGAS(values[i][2]); // C列: 電話番号
        const rowAmount = Number(values[i][3]); // D列: 増減額
        const rowType = String(values[i][4]); // E列: 種別

        let dateMatch = false;
        if (rowDate instanceof Date && data.date) {
          const inputDate = new Date(data.date);
          dateMatch = !isNaN(inputDate.getTime()) && rowDate.getTime() === inputDate.getTime();
        } else {
          dateMatch = String(rowDate) === String(data.date);
        }

        // ★ 電話番号で照合（フォールバック: 名前照合）
        const customerMatch = (editHistPhone && rowPhone) ? (rowPhone === editHistPhone) : (rowName === targetCustomerName);
        if (dateMatch && customerMatch && rowAmount === Number(data.oldAmount) && rowType === data.type) {
          const newAmount = Number(data.newAmount) || 0;
          const diff = newAmount - rowAmount; // 差額（例: 1000→800なら-200）

          // 履歴の金額を更新（D列=4列目）
          historySheet.getRange(i + 1, 4).setValue(newAmount);

          // 残高を差額分調整（★ 電話番号で照合）
          const depSheet = ss.getSheetByName("前払い管理");
          if (depSheet && diff !== 0) {
            const depValues = depSheet.getDataRange().getValues();
            for (let j = 1; j < depValues.length; j++) {
              if ((editHistPhone && normalizePhoneGAS(depValues[j][1]) === editHistPhone) || (!editHistPhone && String(depValues[j][0]).trim() === targetCustomerName)) {
                const currentBal = Number(depValues[j][2]) || 0;
                depSheet.getRange(j + 1, 3).setValue(currentBal + diff);
                break;
              }
            }
          }

          const newBalance = editHistPhone ? getCustomerBalanceByPhone(ss, editHistPhone) : getCustomerBalance(ss, targetCustomerName);
          historySheet.getRange(i + 1, 6).setValue(newBalance); // F列: 残高

          if (lock) lock.releaseLock();
          return output.setContent(JSON.stringify({ success: true, newBalance: newBalance }));
        }
      }

      if (lock) lock.releaseLock();
      return output.setContent(JSON.stringify({ success: false, message: '該当する履歴が見つかりません' }));
    }

    // ⑦ スタッフの追加・マスタ管理
    if (action === 'addStaff') {
      let sheet = ss.getSheetByName("スタッフマスタ");
      if (!sheet) {
        sheet = ss.insertSheet("スタッフマスタ");
        sheet.appendRow(["スタッフ名(ID)", "パスワード", "メールアドレス", "対応サービス"]); 
      }
      sheet.appendRow([data.name, data.password, data.email, data.services || '']);
      return output.setContent(JSON.stringify({ success: true }));
    }

    // ⑦-2 スタッフの編集（パスワード・メアド・対応サービス）
    if (action === 'editStaff') {
      let sheet = ss.getSheetByName("スタッフマスタ");
      if (!sheet) return output.setContent(JSON.stringify({ success: false, message: 'マスタが見つかりません' }));
      const values = sheet.getDataRange().getValues();
      for (let i = 1; i < values.length; i++) {
        if (values[i][0] === data.oldName) {
          if (data.newName !== undefined) sheet.getRange(i + 1, 1).setValue(data.newName);
          if (data.password) sheet.getRange(i + 1, 2).setValue(data.password);
          if (data.email !== undefined) sheet.getRange(i + 1, 3).setValue(data.email);
          if (data.services !== undefined) sheet.getRange(i + 1, 4).setValue(data.services);
          return output.setContent(JSON.stringify({ success: true }));
        }
      }
      return output.setContent(JSON.stringify({ success: false, message: 'スタッフが見つかりません' }));
    }

    // ⑦-3 スタッフの削除
    if (action === 'deleteStaff') {
      let sheet = ss.getSheetByName("スタッフマスタ");
      if (!sheet) return output.setContent(JSON.stringify({ success: false, message: 'マスタが見つかりません' }));
      const values = sheet.getDataRange().getValues();
      for (let i = 1; i < values.length; i++) {
        if (values[i][0] === data.name) {
          sheet.deleteRow(i + 1);
          return output.setContent(JSON.stringify({ success: true }));
        }
      }
      return output.setContent(JSON.stringify({ success: false, message: 'スタッフが見つかりません' }));
    }

    // ⑧ マイページログイン認証
    if (action === 'login') {
      const sheet = ss.getSheetByName("スタッフマスタ");
      if (!sheet) return output.setContent(JSON.stringify({ success: false, message: 'マスタが見つかりません' }));

      const values = sheet.getDataRange().getValues();
      for (let i = 1; i < values.length; i++) {
        // メールアドレス（3列目）とパスワード（2列目）で認証
        if (values[i][2] === data.staffId && String(values[i][1]) === String(data.password)) {
          // 認証成功時にスタッフ名も返す
          return output.setContent(JSON.stringify({ success: true, staffName: values[i][0] }));
        }
      }
      return output.setContent(JSON.stringify({ success: false, message: 'メールアドレスまたはパスワードが違います' }));
    }

    // ⑧-2 顧客マイページログイン認証（電話番号 + パスワード）
    if (action === 'customerLogin') {
      const custSheet = ss.getSheetByName('顧客リスト');
      if (!custSheet) return output.setContent(JSON.stringify({ success: false, message: '顧客リストが見つかりません' }));

      const phone = String(data.phone).replace(/[-\s']/g, '');
      const password = String(data.password);
      const values = custSheet.getDataRange().getValues();

      for (let i = 1; i < values.length; i++) {
        const rowPhone = String(values[i][1]).replace(/[-\s']/g, '');
        const rowPassword = String(values[i][3] || '').trim(); // D列: パスワード
        if (rowPhone === phone && rowPassword === password && rowPassword !== '') {
          return output.setContent(JSON.stringify({
            success: true,
            phone: rowPhone,
            customerName: values[i][0]
          }));
        }
      }
      return output.setContent(JSON.stringify({ success: false, message: '電話番号またはパスワードが正しくありません' }));
    }

    // ⑨ 給与明細メール送信
    if (action === 'sendEmail') {
      try {
        MailApp.sendEmail(data.to, data.subject, data.body);
        return output.setContent(JSON.stringify({ success: true }));
      } catch (e) {
        return output.setContent(JSON.stringify({ success: false, message: 'メール送信失敗: ' + e.toString() }));
      }
    }
    
    // ⑩ テストデータ一括生成（デモ用）
    if (action === 'initDemoData') {
      // シートのリセットと作成
      const sheetsToInit = ['スタッフマスタ', '顧客リスト', 'ブラックリスト', '前払い管理', 'デポジット履歴', '業務報告'];
      sheetsToInit.forEach(sName => {
        let sheet = ss.getSheetByName(sName);
        if (!sheet) {
          sheet = ss.insertSheet(sName);
        } else {
          sheet.clear();
        }
      });

      ss.getSheetByName("スタッフマスタ").appendRow(["スタッフ名(ID)", "パスワード", "メールアドレス", "対応サービス"]);
      ss.getSheetByName("顧客リスト").appendRow(["お客様名", "電話番号", "登録日"]);
      ss.getSheetByName("ブラックリスト").appendRow(["電話番号", "お客様名", "理由", "登録日時"]);
      ss.getSheetByName("前払い管理").appendRow(["お客様名", "電話番号", "現在の残高"]);
      ss.getSheetByName("業務報告").appendRow(["ID", "日付", "スタッフ", "顧客電話番号", "顧客名", "サービス内容", "総売上", "スタッフ報酬", "入金状況", "入金日", "デポジット使用額", "請求額"]);

      // スタッフ10名
      const staffNames = ["吉川", "スタッフA", "スタッフB", "佐藤", "鈴木", "高橋", "田中", "伊藤", "渡辺", "山本"];
      staffNames.forEach(s => ss.getSheetByName("スタッフマスタ").appendRow([s, "mypassword123", "demo@example.com", ""]));

      // 顧客20名
      const customers = [];
      for (let i = 1; i <= 20; i++) {
        customers.push({ name: `テスト顧客${i}号`, phone: `090-${String(i).padStart(4, '0')}-0000` });
        ss.getSheetByName("顧客リスト").appendRow([`テスト顧客${i}号`, `'090-${String(i).padStart(4, '0')}-0000`, todayJST()]);
      }

      // ブラックリスト2名
      ss.getSheetByName("ブラックリスト").appendRow(["'090-9999-9999", "悪質ユーザーA", "未払いが続いたため", nowJST()]);
      ss.getSheetByName("ブラックリスト").appendRow(["'080-8888-8888", "イタズラ迷惑客", "暴言・イタズラ電話", nowJST()]);

      // 前払い3名
      const prepayCust = customers.slice(0, 3);
      prepayCust.forEach((c, idx) => {
        const bl = [10000, 5000, 20000][idx];
        ss.getSheetByName("前払い管理").appendRow([c.name, `'${c.phone}`, bl]);
      });

      // 業務報告30件
      const today = new Date();
      for (let i = 0; i < 30; i++) {
        const rStaff = staffNames[Math.floor(Math.random() * staffNames.length)];
        const rCust = customers[Math.floor(Math.random() * customers.length)];
        const sales = 3000 + Math.floor(Math.random() * 7000);
        const share = Math.floor(sales * 0.5);
        const d = new Date(today);
        d.setDate(today.getDate() - Math.floor(Math.random() * 14)); // 過去2週間
        const isPaid = Math.random() > 0.3 ? '入金済' : '未入金';

        ss.getSheetByName("業務報告").appendRow([
          Utilities.getUuid(),
          d.toLocaleDateString('ja-JP'),
          rStaff,
          `'${rCust.phone}`,
          rCust.name,
          "傾聴(45分 -> 計算45分)",
          sales,
          share,
          isPaid
        ]);
      }

      return output.setContent(JSON.stringify({ success: true, message: 'デモデータの初期化が完了しました。' }));
    }
    
    if (lock) lock.releaseLock();
    return output.setContent(JSON.stringify({ success: false, message: 'Invalid action' }));
  } catch (error) {
    // エラー時もロックを確実に解放
    try {
      const lockActions = ['addReport', 'editReport', 'deleteReport', 'updateDeposit', 'fixDepositBalance', 'deleteDepositHistory', 'editDepositHistory'];
      if (lockActions.includes(JSON.parse(e.postData.contents).action)) {
        const releaseLock = LockService.getDocumentLock();
        if (releaseLock) releaseLock.releaseLock();
      }
    } catch (lockErr) { /* ignore */ }
    return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })).setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const action = e.parameter.action;
  const output = ContentService.createTextOutput().setMimeType(ContentService.MimeType.JSON);

  try {
    // 業務報告一覧の取得
    if (action === 'getReports') {
      const sheet = ss.getSheetByName('業務報告');
      const data = sheet.getDataRange().getValues();
      if (data.length > 0) data.shift();
      // 日付列(index 1)がDate型の場合はJST文字列に変換
      data.forEach(row => {
        if (row[1] instanceof Date) {
          row[1] = Utilities.formatDate(row[1], "JST", "yyyy/MM/dd HH:mm:ss");
        }
      });
      return output.setContent(JSON.stringify({ success: true, data: data }));
    }

    // スタッフリスト取得
    if (action === 'getStaffList') {
      const sheet = ss.getSheetByName("スタッフマスタ");
      const list = [];
      if (sheet) {
        const data = sheet.getDataRange().getValues();
        for(let i=1; i<data.length; i++) {
           if(data[i][0]) list.push({ name: data[i][0], password: data[i][1] || '', email: data[i][2], services: data[i][3] || '' });
        }
      }
      return output.setContent(JSON.stringify({ success: true, staff: list }));
    }

    // ブラックリスト取得
    if (action === 'getBlacklistPhones') {
      const blSheet = ss.getSheetByName('ブラックリスト');
      const phones = [];
      if (blSheet) {
        const blData = blSheet.getDataRange().getValues();
        for (let i = 1; i < blData.length; i++) {
          const phone = String(blData[i][0]).trim();
          if (phone) phones.push(phone);
        }
      }
      return output.setContent(JSON.stringify({ success: true, phones: phones }));
    }

    // 顧客情報とブラックリストの取得
    if (action === 'getCustomerInfo') {
      const result = { success: true, customers: {}, blacklistedPhones: [] };
      
      const custSheet = ss.getSheetByName('顧客リスト');
      if (custSheet) {
        const custData = custSheet.getDataRange().getValues();
        for (let i = 1; i < custData.length; i++) {
          const name = String(custData[i][0]).trim();
          const phone = String(custData[i][1]).trim();
          if (phone && name) {
            result.customers[phone] = name;
          }
        }
      }

      // 万が一リストにない場合のために業務報告の履歴からも名前・電話番号を拾う
      const reportSheet = ss.getSheetByName('業務報告');
      if (reportSheet) {
        const reportData = reportSheet.getDataRange().getValues();
        for (let i = 1; i < reportData.length; i++) {
          const phone = String(reportData[i][3]).trim();
          const name = String(reportData[i][4]).trim();
          if (phone && name && name !== '名無し' && !result.customers[phone]) {
            result.customers[phone] = name;
          }
        }
      }

      const blSheet = ss.getSheetByName('ブラックリスト');
      if (blSheet) {
        const blData = blSheet.getDataRange().getValues();
        for (let i = 1; i < blData.length; i++) {
          const phone = String(blData[i][0]).trim();
          if (phone) result.blacklistedPhones.push(phone);
        }
      }
      return output.setContent(JSON.stringify(result));
    }

    // 残高データの取得（電話番号・パスワード対応）
    if (action === 'getDeposits') {
      const sheet = ss.getSheetByName("前払い管理");
      if (!sheet) return output.setContent(JSON.stringify({ success: false, message: '前払い管理シートがありません' }));

      const data = sheet.getDataRange().getValues();
      const deposits = {};
      const phones = {};
      if (data.length > 1) {
        for (let i = 1; i < data.length; i++) {
          const name = data[i][0];
          const phone = data[i][1];
          const balance = data[i][2];
          if (name) {
            // 重複行がある場合は最初に見つかった行（updateDepositと同じ行）を優先
            if (deposits[name] === undefined) {
              deposits[name] = Number(balance) || 0;
            }
            if(phone && !phones[name]) phones[name] = phone;
          }
        }
      }

      // 顧客リストからパスワードを取得（D列）
      const passwords = {};
      const custSheet = ss.getSheetByName("顧客リスト");
      if (custSheet) {
        const custData = custSheet.getDataRange().getValues();
        for (let i = 1; i < custData.length; i++) {
          const name = custData[i][0];
          const pw = custData[i][3]; // D列: パスワード
          if (name) passwords[name] = String(pw || '');
        }
      }

      return output.setContent(JSON.stringify({ success: true, deposits: deposits, phones: phones, passwords: passwords }));
    }
    // 顧客マイページ用データ取得（残高 + 利用/チャージ履歴）
    if (action === 'getCustomerMyPage') {
      const phone = String(e.parameter.phone || '').replace(/[-\s']/g, '');
      if (!phone) return output.setContent(JSON.stringify({ success: false, message: '電話番号が必要です' }));

      // 顧客名を特定
      let customerName = '';
      const custSheet = ss.getSheetByName('顧客リスト');
      if (custSheet) {
        const custData = custSheet.getDataRange().getValues();
        for (let i = 1; i < custData.length; i++) {
          if (String(custData[i][1]).replace(/[-\s']/g, '') === phone) {
            customerName = custData[i][0];
            break;
          }
        }
      }
      if (!customerName) return output.setContent(JSON.stringify({ success: false, message: '顧客が見つかりません' }));

      // 残高を取得
      let balance = 0;
      const depSheet = ss.getSheetByName('前払い管理');
      if (depSheet) {
        const depData = depSheet.getDataRange().getValues();
        for (let i = 1; i < depData.length; i++) {
          if (depData[i][0] === customerName) {
            balance = Number(depData[i][2]) || 0;
            break;
          }
        }
      }

      // 履歴を構築（デポジット履歴 + 業務報告から利用分）
      const history = [];

      // チャージ履歴（デポジット履歴シートから ★ 電話番号で照合）
      const histSheet = ss.getSheetByName('デポジット履歴');
      if (histSheet) {
        const histData = histSheet.getDataRange().getValues();
        for (let i = 1; i < histData.length; i++) {
          const rowPhone = String(histData[i][2] || '').replace(/[-\s']/g, ''); // C列: 電話番号
          const phoneMatch = phone ? (rowPhone === phone) : (histData[i][1] === customerName);
          if (phoneMatch && histData[i][4] === 'チャージ') { // E列: 種別
            const hDate = histData[i][0] instanceof Date
              ? Utilities.formatDate(histData[i][0], "JST", "yyyy/MM/dd HH:mm:ss")
              : String(histData[i][0]);
            history.push({
              date: hDate,
              type: 'charge',
              chargeAmount: Number(histData[i][3]) || 0, // D列: 増減額
              totalCharged: Number(histData[i][3]) || 0
            });
          }
        }
      }

      // 利用履歴（業務報告シートから）
      const reportSheet = ss.getSheetByName('業務報告');
      if (reportSheet) {
        const repData = reportSheet.getDataRange().getValues();
        for (let i = 1; i < repData.length; i++) {
          if (repData[i][4] === customerName) {
            history.push({
              date: repData[i][1],
              type: 'usage',
              staffName: repData[i][2],
              amount: Number(repData[i][6]) || 0,
              serviceSummary: repData[i][5]
            });
          }
        }
      }

      // 日付の降順でソート
      history.sort(function(a, b) {
        return new Date(b.date) - new Date(a.date);
      });

      return output.setContent(JSON.stringify({
        success: true,
        balance: balance,
        customerName: customerName,
        history: history
      }));
    }

    // デポジット履歴取得
    if (action === 'getDepositHistory') {
      const sheet = ss.getSheetByName("デポジット履歴");
      const history = [];
      if (sheet) {
        const data = sheet.getDataRange().getValues();
        if (data.length > 0) data.shift(); // ヘッダーを削除
        data.forEach(row => {
          // 日付がDate型の場合はJST文字列に変換してから返す
          const dateVal = row[0] instanceof Date
            ? Utilities.formatDate(row[0], "JST", "yyyy/MM/dd HH:mm:ss")
            : String(row[0]);
          history.push({
            date: dateVal,
            customerName: row[1],
            customerPhone: String(row[2] || '').replace(/^'/, ''),
            amount: Number(row[3]) || 0,
            type: row[4],
            balance: Number(row[5]) || 0
          });
        });
      }
      history.reverse();
      return output.setContent(JSON.stringify({ success: true, history: history }));
    }
    
    return output.setContent(JSON.stringify({ success: false, message: 'Invalid action' }));
  } catch (error) {
    return output.setContent(JSON.stringify({ success: false, error: error.toString() }));
  }
}
